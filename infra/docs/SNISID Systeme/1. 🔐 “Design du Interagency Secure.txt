func MatchIdentity(a, b Person) float64 {

    score := 0.0

    if a.Biometrics == b.Biometrics {
        score += 0.6
    }

    if overlap(a.Aliases, b.Aliases) {
        score += 0.3
    }

    return score
}

type MLATPolicy struct {
    CountryA string
    CountryB string
    AllowedData []string
}
📜 VALIDATION
func IsLegal(req AgencyRequest, policy MLATPolicy) bool {

    for _, d := range policy.AllowedData {
        if req.DataType == d {
            return true
        }
    }

    return false
}
