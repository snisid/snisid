🧠 5. AI RISK ENGINE (FINAL LOGIC)
def risk_score(event):
    score = 0

    if event["action"] == "DELETE_USER":
        score += 0.6

    if event["institution"] != event.get("usual_institution"):
        score += 0.3

    return min(score, 1.0)
