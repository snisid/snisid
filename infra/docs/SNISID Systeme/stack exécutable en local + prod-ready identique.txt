🚀 2. ONE COMMAND LOCAL INSTALL (KIND / K3S)
⚙️ bootstrap-kind.sh
#!/bin/bash

set -e

echo "🚀 Creating Kubernetes cluster (kind)..."

kind create cluster --name cyber-civ

echo "📦 Installing base namespaces..."
kubectl create ns soc
kubectl create ns simulation
kubectl create ns gitops

echo "📡 Installing ArgoCD..."
kubectl apply -n gitops -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml

echo "📦 Installing Kafka via Helm..."
helm repo add bitnami https://charts.bitnami.com/bitnami
helm install kafka bitnami/kafka -n simulation

echo "✅ Cluster ready"
