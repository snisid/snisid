🧪 Python inference service
class DeepfakeDetector:
    def predict(self, frame):
        spatial_score = self.cnn(frame)
        freq_score = self.fourier_analyze(frame)
        temporal_score = self.transformer(frame)

        final = (0.4 * spatial_score +
                 0.3 * freq_score +
                 0.3 * temporal_score)

        return {
            "deepfake_score": final,
            "is_fake": final > 0.7
        }
