🚀 2. CI/CD (GITHUB ACTIONS + GITOPS)
🔥 BUILD PIPELINE
name: SNISID CI

on:
  push:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v3

    - name: Build Go services
      run: |
        go build ./services/...

    - name: Docker build
      run: |
        docker build -t snisid/api-gateway ./services/api-gateway-go

    - name: Security scan
      run: |
        trivy image snisid/api-gateway
🚀 DEPLOY PIPELINE (ARGOCD TRIGGER)
name: SNISID Deploy

on:
  workflow_run:
    workflows: ["SNISID CI"]
    types: [completed]

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
    - name: Sync ArgoCD
      run: |
        argocd app sync snisid-government
