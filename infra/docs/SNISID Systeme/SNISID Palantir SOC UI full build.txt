📡 REALTIME STREAM HOOK
export function useSocStream() {
  const [events, setEvents] = useState([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:8090");

    ws.onmessage = (msg) => {
      setEvents(prev => [...prev, JSON.parse(msg.data)]);
    };

    return () => ws.close();
  }, []);

  return events;
}
